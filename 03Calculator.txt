int main() {
    system("color 09");
    printf("Press enter to enter...");

    int usrInput = -1;
    while (usrInput != 0) {
        getchar();
        system("cls");
        draw_main_menu();
        scanf("%d", &usrInput);
        if (usrInput == 1)
            practice_add();
        if (usrInput == 2)
            practice_sub();
        if (usrInput == 3)
            practice_mul();
        if (usrInput == 4)
            practice_div();
        if (usrInput == 5)
            practice_chal();
            getchar();
    }
    exit_message();
}

int create_add() {
    int a, b;
    a = rand() % 10;
    b = rand() % 10;
    printf("\n%d + %d = ?\n", a, b);
    return a + b;
}

int create_sub() {
    int a, b;
    a = rand() % 10;
    b = rand() % 10;
    if (b > a) {
        a = a + b;
        b = a - b;
        a = a - b;
    }
    printf("\n%d - %d = ?\n", a, b);
    return a - b;
}

int create_mul() {
    int a, b;
    a = rand() % 10;
    b = rand() % 10;
    printf("\n%d * %d = ?\n", a, b);
    return a * b;
}

int create_div() {
    int a, b;
    a = rand() % 10 + 1;
    b = rand() % 10 + 1;
    printf("\n%d / %d = ?\n", a * b, a);
    return b;
}

int create_chal() {
    int opt = rand() % 4 + 1;
    int ans;
    switch (opt) {
        case 1:
            ans = create_add();
            break;
        case 2:
            ans = create_sub();
            break;
        case 3:
            ans = create_mul();
            break;
        case 4:
            ans = create_div();
            break;
    }
    return ans;
}

void message(int ctr) {
    if (ctr==0)
        printf("\nWrong answer!");

    if (ctr <= 3)
        printf("\nGreat!");
    else if (ctr <= 6)
        printf("\nGood Job!");
    else if (ctr <= 9)
        printf("\nFantastic!");
    else
        printf("\nIntermediate!");

    printf("\nYou've finish %d problems.", ctr);
}

void practice_add(){
    printf("\nEnter 999 to exit.");
    int ans, usrInput = -1, ctr=0;
    while (usrInput != 999) {
        ans = create_add();
        scanf("%d", &usrInput);
        if (usrInput==999)
            break;
        if (usrInput==ans){
            ctr++;
        } else{
            ctr=0;
        }
        message(ctr);
    }
}
